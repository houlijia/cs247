 97function b = resizeImage(A,m,method,h)
 98% Inputs:
 99%         A       Input Image
100%         m       resizing factor or 1-by-2 size vector
101%         method  'nearest','bilinear', or 'bicubic'
102%         h       the anti-aliasing filter to use.
103%                 if h is zero, don't filter
104%                 if h is an integer, design and use a filter of size h
105%                 if h is empty, use default filter
106
107%if prod(size(m))==1,
108   %bsize = floor(m*size(A));
109%else
110   bsize = m;
111%end
112
113if any(size(bsize)~=[1 2]),
114   error('M must be either a scalar multiplier or a 1-by-2 size vector.');
115end
116
117% values in bsize must be at least 1.
118bsize = max(bsize, 1);
119
120if (any((bsize < 4) & (bsize < size(A))) & ~strcmp(method, 'nea'))
121   fprintf('Input is too small for bilinear or bicubic method;\n');
122   fprintf('using nearest-neighbor method instead.\n');
123   method = 'nea';
124end
125
126if isempty(h),
127   nn = 11; % Default filter size
128else
129   if prod(size(h))==1, 
130      nn = h; h = []; 
131   else 
132      nn = 0;
133   end
134end
135
136[m,n] = size(A);
137
138if nn>0 & method(1)=='b',  % Design anti-aliasing filter if necessary
139
140
141%%%if bsize(1)<m, h1="fir1(nn-1,bsize(1)/m);" else="" end="" if="" bsize(2)<n,="" h2="fir1(nn-1,bsize(2)/n    );" %="" length(h1)="">1 | length(h2)&gt;1, h = h1'*h2; else h = []; end
142
143
144    %if bsize(1)<m,
145        %h=1fir1(nn-1,bsize(1)/m);
146    %else 
147    %end 
148
149    %if bsize(2)<n, 
150        %h2="fir1(nn-1,bsize(2)/n);
151
152    %if length(h1)>1 | length(h2)>1,
153        %h = h1'*h2;
154    %else 
155        %h = []; 
156    %end
157
158
159
160    % ----------------------------------
161    if bsize(1)<m,
162        h1=fir1(10,512/640);    %% LDL place holder ... will not work ... 512/744 will
163        %h1=fir1(nn-1,bsize(1)/m);
164    else 
165        h1=1
166        %h1=[] % LDL added
167    end 
168
169    if bsize(2)<n,
170        h2=fir1(10,512/640);    %% LDL 
171        %h2=fir1(nn-1,bsize(2)/n); 
172    else
173        h2=1
174        %h2=[] % LDL added
175    end
176
177    if length(h1)>1 | length(h2)>1, 
178        h = h1'*h2; 
179    else 
180        h = []; 
181    end
182
183    % ----------------------------------
184
185
186
187
188
189
190
191
192
193
194
195
196
197
198
199
200
201
202   if length(h1)>1 | length(h2)>1, 
203      a = filter2(h1',filter2(h2,A)); 
204   else 
205      a = A; 
206   end
207elseif method(1)=='b' & (prod(size(h)) > 1),
208   a = filter2(h,A);
209else
210   a = A;
211end
212
213if method(1)=='n', % Nearest neighbor interpolation
214   dx = n/bsize(2); dy = m/bsize(1); 
215   uu = (dx/2+.5):dx:n+.5; vv = (dy/2+.5):dy:m+.5;
216elseif all(method == 'bil') | all(method == 'bic'),
217   uu = 1:(n-1)/(bsize(2)-1):n; vv = 1:(m-1)/(bsize(1)-1):m;
218else
219   error(['Unknown interpolation method: ',method]);
220end
221
222%
223% Interpolate in blocks
224%
225nu = length(uu); nv = length(vv);
226%blk = bestblk([nv nu]);
227blk = [ 64 64 ] % LDL added
228nblks = floor([nv nu]./blk); nrem = [nv nu] - nblks.*blk;
229mblocks = nblks(1); nblocks = nblks(2);
230mb = blk(1); nb = blk(2);
231
232rows = 1:blk(1); b = zeros(nv,nu);
233for i=0:mblocks,
234
235   if i==mblocks,
236      rows = (1:nrem(1));
237   end
238  
239   cols = 1:blk(2); 
240   for j=0:nblocks,
241
242      if j==nblocks, 
243        cols=(1:nrem(2));
244      end
245
246      if ~isempty(rows) & ~isempty(cols)
247         [u,v] = meshgrid(uu(j*nb+cols),vv(i*mb+rows));
248         % Interpolate points
249         %if method(1) == 'n', % Nearest neighbor interpolation
250            %b(i*mb+rows,j*nb+cols) = interp2(a,u,v,'*nearest');
251         %elseif all(method == 'bil'), % Bilinear interpolation
252            %b(i*mb+rows,j*nb+cols) = interp2(a,u,v,'*linear');
253         %elseif all(method == 'bic'), % Bicubic interpolation
254            b(i*mb+rows,j*nb+cols) = interp2(a,u,v,'cubic');
255         %end
256      end
257   end
258end
259
260if nargout==0,
261   %%% LDL if isgray(b), imshow(b,size(colormap,1)), else imshow(b), end
262   % LDL imshow(b)
263   return
264end
265
266%%% LDL if isgray(A)   % This should always be true
267   b = max(0,min(b,1));  
268%%% LDL end
